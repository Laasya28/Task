# MongoDB Compass Connection String
# mongodb://admin:password@localhost:27017/student-task-manager?authSource=admin

# Frontend: http://localhost:3000
# Backend API: http://localhost:5000/api

# Environment Variables for Backend
# MONGODB_URI=mongodb://admin:password@mongodb:27017/student-task-manager?authSource=admin
# JWT_SECRET=your-secret-key-change-this-in-production
# PORT=5000
# NODE_ENV=development
# FRONTEND_URL=http://localhost:3000

# To run:
# docker-compose up --build

# To stop:
# docker-compose down

# To view logs:
# docker-compose logs -f

# To access MongoDB in container:
# docker exec -it task-manager-mongodb mongosh -u admin -p password
